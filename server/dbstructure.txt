Designing the Database

Static Data: 
	- product table
	- discount table
	- user table
	- feedback table

Session Data: 
	- shopping_session table
	- cart_item table

Processed Data: 
	- order_details table
	- order_items table
	- payment_details table

///// Static Data
users: 
	- id: uuid not null/ primary key
	- user_name: text not null
	- type: text default null
	- oauth_id: text default null
	- email: text unique default null
	- phone_number: text unique default null
	- password: text default null
	- avatar_url: text default null
	- address: text default null
	- created_at: time

shops: 
	- id: uuid not null / primary key
	- name: string not null
	- user_id: uuid not null / foreign key users.id
	- followers: int default 0
	- following: int default 0

products: 
	- id: uuid not null / primary key
	- name: string not null (index)
	- desc: text default null
	- price: decimnal not null
	- category: string not null
	- shop_id: uuid not null / foreign key shops.id
	- discount_id: uuid default null / foreign key to discount.id
	- feedback_id: uuid default null / foreign key to feedback.id
	- rating: decimal default null

messages: 
	- id: uuid not null
	- chat: text not null
	- user_id: uuid not null / foreign key users.id
	- text: text default null
	- photo: text default null

feedbacks: 
	- id: uuid not null/ primary key
	- text: text default null
	- photo: string(999) default null
	- rating: decimal not null

discounts: 
	- id: uuid not null/ primary key
	- type: string not null ['app', 'shop', 'category', 'product']
	- desc: text default null
	- discount_percent: integer default null
	- discount: decimal default null
	- quantity: int default null
	- min_price: decimal default null
	- time_apply: time not null
	- time_expried: time not null

my_vouchers: 
	- user_id: uuid not null / foreign key users.id
	- discount_id: uuid not null / foreign key discounts.id
	- used: boolean default false

///// Session Data

carts:
	- id: uuid not null
	- user_id: uuid not null / foreign key users.id
	- amount: decimal not null
	- session: boolean deafault true

cart_items: 
	- cart_id: uuid not null / foreign key carts.id
	- product_id: uuid not null
	- quantity: int not null

///// Processed Data

order_details: 
	- id: uuid not null
	- user_id: uuid not null / foreign key users.id 
	- quantity: int not null 
	- total: decimal not null
	- app_voucher: uuid default null
	- shop_voucher: uuid default null
	- free_shpping_voucher: uuid default null
	- status: string [null, 'to pay', 'to ship', 'to receive', 'completed', 'cancelled', 'return refund']

order_items: 
	- order_id: uuid not null / foreign key order_details.id
	- product_id: uuid not null / foreign key products.id
	- created_at: time
	- updated_at: time
